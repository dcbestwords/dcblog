隐藏搜索栏
<template>
    <div class="bc">
        <div class="search" :class="{active:isActive}">
            <input type="text" class="input" placeholder="Search...">
            <button @click="changeState" class="btn">
                <i class="fas fa-search"></i>
            </button>
        </div>
    </div>
</template>

<script>
export default {
    name: 'HiddenSearch',
    data() {
        return {
            isActive:false
        }
    },
    methods: {
        changeState(){
            this.isActive = !this.isActive
        }
    },
}
</script>
